# 🎈 class declaration
```js
'use strict'
class Person{
    // constructor
    constructor(name, age){
        //fields
        this.name = name;
        this.age = age;
    }

    // method
    speak(){
        console.log(`${this.name}(${this.age}): hola!`);
    }
}

// object
const dori = new Person('dori', '100');

console.log(dori.name);
console.log(dori.age);
dori.speak();
```
> dori
        100
        dori(100): hola!

# 🎈 getter & setter
```js
class Emp{
    constructor(name, age, job){
        this.name = name;
        this.age = age;
        this.job = job;
    }

    get age(){
        return this._age;
    }

    set age(value){
        this._age = value < 0 ? 0 : value;
    }
}

const emp1 = new Emp('dori', -100, 'frontend developer');
console.log(`${emp1.name}(${emp1.age}) is ${emp1.job}`);
```
> dori(0) is frontend developer
